//A function to record how long the compressor has been on 
var startTime = new Date, endTime = new Date, time_diff = 0, prev_signal = 0 ; 
function ontime (signal){
  if (prev_signal = 0) {
    if (signal = 0){
      prev_signal = 0;
      return 0;
    } else {
      startTime = Date.now();
      prev_signal = 1;
      return 0;
    }
  } else {
    if (signal = 0){
      prev_signal = 0;
      return 0;
    } else {
      endTime = Date.now();
      time_diff = Math.round((endTime - startTime)/1000) ;
      prev_signal = 1;
      return (time_diff);
    }
  }    
}
